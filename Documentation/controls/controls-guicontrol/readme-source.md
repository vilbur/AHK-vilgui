# Controls GuiControl  ## Methods  __edit__( string|int|boolean __$value__ ) @return object __Control__  	// set value of control  __guiControl__( string __$command__, [ mixin __$value__ ] ) @return object __Control__  	// wrapper for [guiControl](https://autohotkey.com/docs/commands/GuiControl.htm)  __size__( int __$width__, [ int __$height__ ] ) @return object __Control__  	// set size of control  __pos__( [ int __$x__, int __$y__ ] ) @return object {x,y}|__Control__  	// __set__ position of control or __get__ position if parameters are empty  ## Examples
